
### 二叉树的后序遍历
左子树 -〉 右子树 -〉根节点
```go
func postorderTraversal(root *TreeNode) []int {
    if root == nil {
        return nil
    }

    stack := []*TreeNode{root}
    var prev *TreeNode
    var res []int

    for len(stack) > 0 {
        top := stack[len(stack)-1]
        if (top.Right == nil || top.Right == prev) && (top.Left == nil || top.Left == prev) {
            stack = stack[:len(stack)-1]
            res = append(res, top.Val)
            prev = top
        } else {
            if top.Right != nil {
                stack = append(stack, top.Right)
            }
            if top.Left != nil {
                stack = append(stack, top.Left)
            }
        }
    }

    return res
}
```
